cmake_minimum_required(VERSION 3.20)

set(CMAKE_CXX_SCAN_FOR_MODULES OFF)
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

project(DOMK VERSION 0)

set(GLFW3_ROOT "C:/Users/USER/include")
set(GLFW3_LIB_ROOT "C:/Users/USER/libs")
set(GLM_ROOT_DIR "C:/Users/USER/include")
set(GLM_LIB_ROOT "C:/Users/USER/libs")

find_package(Vulkan REQUIRED)
find_package(GLFW3 REQUIRED)
find_package(GLM REQUIRED)
find_package(tinyobjloader REQUIRED)

file(GLOB src "src/*.cpp" "src/*.c")

add_executable(DOMK ${src})

target_compile_features(DOMK PRIVATE cxx_std_23)

target_include_directories(DOMK PUBLIC "include")
target_include_directories(DOMK PUBLIC ${GLFW3_INCLUDE_DIR})
target_include_directories(DOMK PUBLIC ${GLM_INCLUDE_DIR})
target_include_directories(DOMK PUBLIC ${Vulkan_INCLUDE_DIRS})

target_link_libraries(DOMK PUBLIC Vulkan::Vulkan)
target_link_libraries(DOMK PUBLIC ${GLM_LIB_PATH})
target_link_libraries(DOMK PUBLIC ${GLFW3_LIBRARY})
